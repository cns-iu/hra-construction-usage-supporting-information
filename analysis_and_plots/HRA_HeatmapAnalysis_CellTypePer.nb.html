HRA-Pop Cell Type Correlation Analysis Notebook

library(tidyr)
library(dplyr)
library(stringr)
library(magrittr)
options(scipen=999)
data <- read.csv("https://raw.githubusercontent.com/x-atlas-consortia/hra-pop/main/output-data/v0.9.0/reports/atlas/validation-v5.csv",
                 header=T)
data$organ <- data$organ %>%
  str_replace("left ", "") %>%
  str_replace("right ", "") %>%
  str_replace("Set of ", "") %>%
  str_replace("left ", "") %>%
  str_replace("In ", "in")

# Remove spatial data
data <- data[data$modality!="sc_proteomics",]

Azimuth

# Create pivot
data_ha <- data %>% 
              filter(tool == "azimuth") %>%
              select(organ, cell_id, percentage) %>%
              pivot_wider(names_from = cell_id,
                          values_from = percentage,
                          values_fill = 0,
                          values_fn = mean)

# Convert DF to a matrix
data_hap <- data.matrix(data_ha[,2:94])

# Assign Row.names as organs
rownames(data_hap) <- data_ha$organ 
data_hap <- data_hap[order(row.names(data_hap)),]

# Scale data
data_hap <- scale(data_hap)

# Generate heatmap
pheatmap::pheatmap(data_hap)

PopV

# Create pivot
data_hp <- data %>% 
              filter(tool == "popv") %>%
              select(organ, cell_id, percentage) %>%
              pivot_wider(names_from = cell_id,
                          values_from = percentage,
                          values_fill = 0,
                          values_fn = mean)

# Convert DF to a matrix
data_hpv <- data.matrix(data_hp[,2:87])

# Assign Row.names as organs
rownames(data_hpv) <- data_hp$organ 
data_hpv <- data_hpv[order(row.names(data_hpv)),]

# Scale data
data_hpv <- scale(data_hpv)

# Generate heatmap
pheatmap::pheatmap(data_hpv)
